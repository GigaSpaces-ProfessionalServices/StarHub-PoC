* CreateStaticRouting

  configure routerstatic
    edit 1
        set dst  <address and subnet to reach>
        set gateway  <gateway ip address>
        set device port <port no>
  end

* Create Firewall Address

  configure firewall address
    edit rule1
        set subnet <ip address> <mask>
        set associated‐interface port <port no>
  end


* Create Firewall Service

  config firewall service custom
     edit firewallServer
         set protocol tcp
         set tcp‐portrange <port no>
  end


* Create Firewall Policy

  configure firewall policy
    edit <policy id>
        set srcintf port <port no>
        set dstintf port <port no>
        set srcaddr <based on the above firewall address>
        set dstaddr <based on the above firewall address>
        set service <based on the above firewall service>
        set action  <accept/deny>
